fun int main()
    var matrix m;
    var int fila,columna,total;
    m = new int[2][3] {1, 2, 3, 4, 5, 6};
    fila = 0;
    total = 0;
    while fila < 2 do
        columna = 0;
        while columna < 3 do
            total = total + m[fila][columna];
            columna = columna + 1
        endwhile;
        fila = fila + 1
    endwhile;
    print(total);
    return(0)
endfun
